def create_rfm_adaptor_logger
  log_dir = File.join(RfmAdaptor.root, "log")
  FileUtils.mkdir_p(log_dir)
  log_path = File.join(log_dir, "rfm_adaptor.log")
  FileUtils.touch(log_path)
  Logger.new(log_path)
end
